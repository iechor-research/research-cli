/**
 * 初始化研究工具模块
 * 
 * 这个函数会：
 * 1. 注册所有可用的研究工具
 * 2. 设置默认配置
 * 3. 准备工具注册中心
 */
// 添加一个全局标志防止重复初始化
let globalInitialized = false;
 
export function initializeResearchTools(): void {
  if (globalInitialized) {
    return;
  }
  globalInitialized = true;
 
  // 延迟初始化研究工具（避免循环依赖）
  import('./init.js').then(async (initModule) => {
    const { ResearchToolRegistry } = await import('./registry.js');
    initModule.initializeResearchTools(ResearchToolRegistry.getInstance());
  }).catch(console.error);
}
 
/**

export function getModuleInfo() {

    author: 'iEchor LLC',
  };
}
 
/**
 * 检查研究工具模块的健康状态
 */
export function checkModuleHealth() {
  try {
    return {
      status: 'healthy' as const,
      message: 'Research tools module is functioning normally',
      timestamp: new Date().toISOString(),
    };
  } catch (error) {
    return {
      status: 'error' as const,
      message: error instanceof Error ? error.message : 'Unknown error in research tools module',
      timestamp: new Date().toISOString(),
    };
  }
}
